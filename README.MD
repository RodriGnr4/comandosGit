# comandos utiles de git

1. git init (inicia repositorio)
2. git add <archivo>
3. git reset <archivo>
4. git commit -m
5. git checkout -- .
6. git log
7. git commit --amend
8. git checkout -b nombreRama
9. git checkout master
10. git merge
11. git branch -d nombreRama
12. git branch -M main
13. git push -u origin main
14. git commit -am
